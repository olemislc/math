\begin{problem}
	Halle todas las funciones  continuas $f : \mathbb{R}\textsuperscript{+} \rightarrow \mathbb{R}\textsuperscript{+}$ tales que $\forall x \in \mathbb{R}\textsuperscript{+}$
	\begin{equation}
		f(x+1) = f(x) + 1
	\end{equation}
	\begin{equation}
		f\left(\frac{1}{f(x)}\right) = \frac{1}{x}
	\end{equation}
\end{problem}
