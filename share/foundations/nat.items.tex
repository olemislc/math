\begin{axiom} \label{nat_zero}
	Elemento nulo
	\begin{equation}
		0 \in \mathbb{N}
	\end{equation}
\end{axiom}
\begin{axiom} \label{nat_closure}
	El conjunto de los números naturales es cerrado con respecto a la relación de sucesor.
	\begin{equation}
		x \in \mathbb{N}\ \implies S(x) \in \mathbb{N}
	\end{equation}
\end{axiom}
