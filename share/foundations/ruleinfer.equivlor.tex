\begin{ruleinf} \label{relinf_equivlor} Eliminación bicondicioal por disjunción
	\begin{equation}
	\begin{gathered}
		\varphi \leftrightarrow \psi \\
		\underline {\psi \lor \varphi } \\
		\psi \land \varphi 
	\end{gathered}
	\end{equation}
	\begin{equation}
	\begin{gathered}
		\varphi \leftrightarrow \psi \\
		\underline {\neg\psi \lor \neg\varphi } \\
		\neg\psi \land \neg\varphi 
	\end{gathered}
\end{equation}
\end{ruleinf}
