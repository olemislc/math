\begin{axiom} \label{real_lub}
	Axioma del supremo
	\begin{equation} \label{eq_real_lub}
	\begin{gathered}
		A \subset \mathbb{R}, A \neq \emptyset, \\
		\exists v \in \mathbb{R}\ \forall x \in A\ x \leq v\ \vdash\ \exists u \in A\ (u \leq v) \land (\forall x \in A\ x \leq u)
	\end{gathered}
	\end{equation}
\end{axiom}
