\begin{ruleinf} \label{relinf_cases} Currying
	\begin{equation}
	\begin{gathered}
		\underline {\varphi, \phi  \vdash \psi } \\
		\varphi \vdash \phi \to \psi 
	\end{gathered}
	\end{equation}

	\begin{equation}
		\begin{gathered}
			\underline {\varphi \vdash \phi \to \psi } \\
			\varphi, \phi  \vdash \psi 
		\end{gathered}
	\end{equation}
\end{ruleinf}
