\begin{ruleinf} \label{relinf_absurdum} Reductio ad absurdum
	\begin{equation}
	\begin{gathered}
	    \varphi \vdash \psi \\
		\underline {\varphi \vdash \lnot \psi } \\
		\lnot \varphi 
	\end{gathered}
	\end{equation}
	\begin{equation}
	\begin{gathered}
		\lnot\varphi \vdash \psi \\
		\underline {\lnot\varphi \vdash \lnot \psi } \\
		\varphi 
	\end{gathered}
\end{equation}
\end{ruleinf}
